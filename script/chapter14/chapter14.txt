【示例14-3】调用过程分页查询员工表
本示例调用过程MYPACK.Get_EmployeeByPage_P查询员工表的第1页(每页3个员工)。
SQL> 
set serveroutput on
declare
rc sys_refcursor;
lrow employees%rowtype; 
begin
MYPACK.Get_EmployeeByPage_P(1，3，rc);    
    loop
      FETCH rc INTO lrow;
      exit when rc%notfound;      
      DBMS_OUTPUT.put_line (lrow.employee_id || '  '|| lrow.name);
    end loop;
    CLOSE rc;    --关闭游标
 end;
 /
